Mapfold

09 Apr 2014 — posted in programming

Ich habe lange kein Erlang mehr geschrieben, jetzt kommt es in einer Vorlesung dran. Als Fingerübung (und zum Testen des Code-Highlightings hier) eine endrekursive mapfold-Implementation.

mapfold(Func, Accu, List) ->
    mapfold(Func, Accu, [], List).

mapfold(Func, Accu, LiAccu, [Head|Tail]) ->
    {NewElem,NewAccu} = Func(Head,Accu),
    mapfold(Func,NewAccu,[NewElem|LiAccu],Tail);
mapfold(_, Accu, LiAccu, []) ->
    {lists:reverse(LiAccu),Accu}.
